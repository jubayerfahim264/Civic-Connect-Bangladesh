// Firestore Security Rules for CivicConnect
// Deploy these rules in Firebase Console: https://console.firebase.google.com/project/{projectId}/firestore/rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection: each user can only access their own document
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Services collection: authenticated users can read, only admins can write
    match /services/{serviceId} {
      allow read: if request.auth != null;
      allow create, update, delete: if request.auth != null && isAdmin(request.auth.uid);
    }
    
    // Helper function to check if user is admin
    function isAdmin(uid) {
      return exists(/databases/$(database)/documents/users/$(uid)) &&
             get(/databases/$(database)/documents/users/$(uid)).data.role == 'admin';
    }
    
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

/* 
DEPLOYMENT INSTRUCTIONS:
1. Go to Firebase Console: https://console.firebase.google.com
2. Select your project (civicconnect-bangladesh)
3. Navigate to Firestore Database > Rules
4. Copy and paste the rules above
5. Click "Publish"

IMPORTANT NOTES:
- Each user can ONLY read/write their own document in /users/{userId}
- All authenticated users can READ the services collection
- Only users with role == 'admin' can CREATE/UPDATE/DELETE services
- To make a user an admin, add a "role" field with value "admin" to their user document
- All other collections are blocked by default

TESTING:
1. Sign up a new user - their user document is created automatically
2. Try to access another user's profile - should be blocked
3. Try to add a service without admin role - should be blocked
4. Sign in as admin user - should be able to modify services
